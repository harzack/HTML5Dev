Exemple 6-2 : Pour supprimer le groupe « Gestionnaires » de la liste de contrôle d'accès de l'objet « Nouveau dossier » : 

<import>
  <node type="folder" action="update">
    <location>Enterprise:New Folder</location>
    <acl group="Managers" action="remove"/>
  </node>
</import>
